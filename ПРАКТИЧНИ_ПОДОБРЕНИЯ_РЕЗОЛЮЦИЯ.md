```typescript
// Добави в src/components/screens/ImageUploadScreen.tsx

const checkImageQuality = (imageDataUrl: string): Promise<{
  width: number
  height: number
  isGoodQuality: boolean
  warning?: string
}> => {
  return new Promise((resolve) => {
    const img = new Image()
    img.onload = () => {
      const isGoodQuality = img.width >= 800 && img.height >= 800
      const warning = !isGoodQuality 
        ? `Изображението е ${img.width}x${img.height}px. За най-добри резултати, моля качете снимка поне 800x800px.`
        : undefined
      
      resolve({
        width: img.width,
        height: img.height,
        isGoodQuality,
        warning
      })
    }
    img.src = imageDataUrl
  })
}

// Използвай в handleCropComplete:
const quality = await checkImageQuality(croppedDataUrl)
if (quality.warning) {
  // Покажи warning toast на потребителя
  addLog('warning', quality.warning)
}
```

### Helper функция за detail crop:

```typescript
const createDetailCrop = (
  dataUrl: string,
  options: { targetSize: number; quality: number; centerFocus: boolean }
): Promise<string> => {
  return new Promise((resolve, reject) => {
    const img = new Image()
    img.onload = () => {
      const canvas = document.createElement('canvas')
      const ctx = canvas.getContext('2d')
      if (!ctx) {
        reject(new Error('No canvas context'))
        return
      }
      
      // Кропни центъра
      const size = Math.min(img.width, img.height)
      const sourceX = (img.width - size) / 2
      const sourceY = (img.height - size) / 2
      
      canvas.width = options.targetSize
      canvas.height = options.targetSize
      
      // Изрежи и преоразмери
      ctx.drawImage(
        img,
        sourceX, sourceY, size, size,  // source
        0, 0, options.targetSize, options.targetSize  // destination
      )
      
      resolve(canvas.toDataURL('image/jpeg', options.quality))
    }
    img.onerror = reject
    img.src = dataUrl
  })
}
```

### Очакван резултат:
- ✅ AI вижда и "голямата картина" и детайли
- ✅ Само 2x изображения (не 5x като Grid & Tile)
- ✅ Без нужда от Python backend
- ⚠️ 2x API calls (все пак по-евтино от 5x)

### Време за имплементиране: **2-3 часа**
